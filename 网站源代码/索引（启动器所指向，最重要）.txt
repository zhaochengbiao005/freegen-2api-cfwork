    function injectAdOnce() {
        // Only inject if enabled, we've had more than the threshold generations, and ad not already injected
        if (!GENERATION_AD_CONFIG.enabled || inlineAdGenerationCount < GENERATION_AD_CONFIG.showAfterGeneration || adAlreadyInjected) {
            return;
        }

        console.log('Injecting permanent generation ad');

        // Fire GTM event for AdSense (if using AdSense)
        if (GENERATION_AD_CONFIG.type === 'adsense') {
            window.dataLayer = window.dataLayer || [];
            window.dataLayer.push({
                'event': 'image_generation_ad_injected',
                'generation_count': inlineAdGenerationCount
            });
        }

        // Handle BuySellAds type
        if (GENERATION_AD_CONFIG.type === 'buysellads') {
            // Create ad container with styles
            const adContainer = document.createElement('div');
            adContainer.className = 'generation-ad-container mt-4 text-center';
            adContainer.style.marginBottom = '250px';
            adContainer.innerHTML = `
                <style>
                    .generation-ad-container {
                        display: flex;
                        justify-content: center;
                        max-width: 100%;
                    }
                </style>
                <div id="bsa-zone_1768997875379-6_123456"></div>
            `;

            // Insert after the entire section (outside the 3-column layout)
            const section = imageContainer.closest('section');
            if (section) {
                section.parentNode.insertBefore(adContainer, section.nextSibling);
            } else {
                // Fallback: insert after image container's row
                const row = imageContainer.closest('.row');
                if (row) {
                    row.parentNode.insertBefore(adContainer, row.nextSibling);
                } else {
                    imageContainer.parentNode.insertBefore(adContainer, imageContainer.nextSibling);
                }
            }

            // Use Optimize Push function to display the ad
            window.optimize = window.optimize || { queue: [] };
            window.optimize.queue.push(function() {
                window.optimize.push('bsa-zone_1768997875379-6_123456');
            });

            adAlreadyInjected = true;
            console.log('BuySellAds generation ad injected');
            return;
        }

        // Load the ad template for other types
        fetch('/templates/partials/generation-ad.html')
            .then(response => response.text())
            .then(html => {
                // Create ad container
                const adContainer = document.createElement('div');
                adContainer.innerHTML = html;

                // Insert after image container
                imageContainer.parentNode.insertBefore(adContainer.firstElementChild, imageContainer.nextSibling);

                // If AdSense, push the ad
                if (GENERATION_AD_CONFIG.type === 'adsense' && window.adsbygoogle) {
                    try {
                        (adsbygoogle = window.adsbygoogle || []).push({});
                        console.log('AdSense ad pushed for generation', inlineAdGenerationCount);
                    } catch(e) {
                        console.error('AdSense push failed:', e);
                    }
                }

                adAlreadyInjected = true;
                console.log('Permanent generation ad injected (type:', GENERATION_AD_CONFIG.type, ')');
            })
            .catch(err => {
                console.error('Failed to load generation ad:', err);
            });
    }

    function setButtonLoading() {
      if (!originalBtnHTML) {

    // Function to create WebSocket authentication token
    async function createWebSocketAuth(jobId) {
      const timestamp = Math.floor(Date.now() / 1000);
      const message = jobId + timestamp;

      // Simple auth for development (in production, use proper HMAC)
      
      // In production, this should use the same HMAC as the signer
      const encoder = new TextEncoder();
      const data = encoder.encode(message);
      const hashBuffer = await crypto.subtle.digest('SHA-256', data);
      const hashArray = Array.from(new Uint8Array(hashBuffer));
      const hashHex = hashArray.map(b => b.toString(16).padStart(2, '0')).join('');
      return btoa(hashHex).substring(0, 20) + ':' + timestamp;
      
    }

    // Function to resize an image from a data URL using canvas
    function resizeImage(dataUrl, maxSize) {
      return new Promise((resolve, reject) => {
        const img = new Image();
        img.onload = function () {
          let { width, height } = img;
          if (width > height) {
            if (width > maxSize) {
              height = Math.round(height * (maxSize / width));
              width = maxSize;
            }
          } else {
            if (height > maxSize) {
              width = Math.round(width * (maxSize / height));
              height = maxSize;
            }
          }
          const canvas = document.createElement("canvas");
          canvas.width = width;
          canvas.height = height;
          const ctx = canvas.getContext("2d");
          ctx.drawImage(img, 0, 0, width, height);
          const resizedDataUrl = canvas.toDataURL("image/jpeg", 0.7);
          resolve(resizedDataUrl);
        };
        img.onerror = function(err) {
          reject(err);
        };
        img.src = dataUrl;
      });
    }

    async function generateImage() {
      // Prevent multiple concurrent jobs
      if (currentJobId && !jobCompleted) {
        errorContainer.textContent = 'Please wait for the current job to complete.';
        return;
      }

      errorContainer.textContent = '';

      // Clear previous image
      imageContainer.innerHTML = '';

      let prompt = promptInput.value.trim();
      const imageFile = imageInput.files[0];

      if (!prompt && !imageFile) {
        errorContainer.textContent = 'Please enter a prompt.';
        return;
      }
      if (!prompt) {
        prompt = '.';
      }
      const selectedRatio = document.getElementById('aspectRatioSelect')?.value || "4:3";

      if (prompt.length > 2000) {
        errorContainer.textContent = 'Prompt must be less than 2000 characters.';
        return;
      }

      // Set button to loading state
      setButtonLoading();

      // Inject ad immediately on first generation (before waiting for result)
      inlineAdGenerationCount++;
      localStorage.setItem('inlineAdGenerationCount', inlineAdGenerationCount.toString());
      injectAdOnce();

      // Hide SEO content sections and scroll to focus on generation
      hideSeoSections();

      // Smooth scroll to the generation area
      setTimeout(() => {
        promptInput.scrollIntoView({ behavior: 'smooth', block: 'center' });
      }, 100);

      try {
        // Get signed data from signer
        const signerRes = await fetch(SIGNER_URL, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ prompt })
        });

        // Check for errors on the signer response
        if (!signerRes.ok) {
          if (signerRes.status === 429) {
            throw new Error('Too many requests. Please wait a moment and try again.');
          } else {
            throw new Error(`Signer Error (${signerRes.status})`);
          }
        }

        const { ts, sig } = await signerRes.json();

        // Prepare the body for generator request
        const body = {
          prompt,
          ts,
          sig,
          ratio_id: selectedRatio,
        };

        if (imageFile && resizedImageData) {
          body.image_data = resizedImageData;
        }

        // Request image generation
        const generatorRes = await fetch(GENERATOR_URL, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify(body)
        });

        // Check for errors on the generator response
        if (!generatorRes.ok) {
          // Try to parse the error response
          let errorData;
          try {
            errorData = await generatorRes.json();
          } catch {
            errorData = { error: `Generator Error (${generatorRes.status})` };
          }

          // Check if it's an image refusal error
          if (generatorRes.status === 400 && errorData.user_error) {
            throw new Error(errorData.error || 'Image cannot be processed. Please try a different image.');
          } else if (generatorRes.status === 429) {
            throw new Error('Too many requests. Please wait a moment and try again.');
          } else {
            throw new Error(errorData.error || `Generator Error (${generatorRes.status})`);
          }
        }

        const data = await generatorRes.json();

        // Track job start time for performance metrics (for both WebSocket and fallback)
        const jobStartTime = Date.now();

        // Check if we got a job_id (new queue-based system)
        if (data.job_id) {
          // Track this job and sync button state
          currentJobId = data.job_id;
          jobCompleted = false;
          syncButtonState(); // Ensure button stays in loading state

          // Connect to WebSocket for real-time updates
          const auth = await createWebSocketAuth(data.job_id);
          const ws = new WebSocket(WEBSOCKET_URL);

          let wsConnected = false;
          let wsTimeout = null;

          ws.onopen = () => {
            wsConnected = true;
            // Subscribe to this job
            ws.send(JSON.stringify({
              type: 'subscribe',
              job_id: data.job_id,
              auth: auth
            }));

            // Set timeout for connection (5 minutes)
            wsTimeout = setTimeout(() => {
              if (ws.readyState === WebSocket.OPEN) {
                ws.close();
                errorContainer.textContent = 'Connection timeout. Please try again.';
                // Mark job as completed and sync button state on timeout
                jobCompleted = true;
                currentJobId = null;
                syncButtonState();
              }
            }, 300000);
          };

          ws.onmessage = (event) => {
            const message = JSON.parse(event.data);

            if (message.type === 'status') {
              // Update status in UI (optional - could add a status display)
              console.log('Status:', message.message);
            } else if (message.type === 'result') {
              // Display the generated image
              const imageUrl = message.image_data;
              const truncatedPrompt = prompt.substring(0, 900);
              const encodedPrompt = encodeURIComponent(truncatedPrompt).replace(/%20/g, '-');
              const findSimilarBtn = (prompt && prompt.trim() && prompt.trim() !== '.') ?
                `<a href="https://imgsearch.com/search/${encodedPrompt}?ref=freegen" target="_blank" rel="nofollow" class="btn btn-outline-secondary mt-2 me-2 simple-btn">Find Similar</a>` :
                '';
              imageContainer.innerHTML = `
                <img src="${imageUrl}" alt="Generated Image" class="img-fluid">
                <br>
                ${findSimilarBtn}
                <a href="${imageUrl}" download="generated_image.jpg" class="btn btn-primary mt-2">Download Image</a>
              `;

              // Mark job as completed and sync button state
              jobCompleted = true;

              // Record completion timing metrics
              const totalTime = Date.now() - jobStartTime;
              console.log('Recording completion time:', totalTime, 'ms for job:', data.job_id);
              fetch(`${STATS_API_URL}/record-completion`, {
                method: 'POST',
                headers: {'Content-Type': 'application/json'},
                body: JSON.stringify({
                  job_id: data.job_id,
                  total_time_ms: totalTime,
                  timestamp: new Date().toISOString()
                })
              })
              .then(res => res.json())
              .then(result => console.log('Timing recorded:', result))
              .catch(err => console.error('Failed to record timing metrics:', err));

              // Increment generation count for ad tracking (popup ads)
              incrementGenerationCount();

              currentJobId = null;
              if (wsTimeout) clearTimeout(wsTimeout);
              syncButtonState();
              ws.close();
            } else if (message.type === 'error') {
              errorContainer.textContent = message.message || 'Failed to generate image.';
              imageContainer.innerHTML = '';

              // Mark job as completed and sync button state
              jobCompleted = true;
              currentJobId = null;
              if (wsTimeout) clearTimeout(wsTimeout);
              syncButtonState();
              ws.close();
            }
          };

          ws.onerror = (error) => {
            console.error('WebSocket error:', error);
            if (!wsConnected) {
              errorContainer.textContent = 'Failed to connect for real-time updates. Please try again.';
              // Mark job as completed and sync button state on WebSocket connection failure
              jobCompleted = true;
              currentJobId = null;
              syncButtonState();
            }
          };

          ws.onclose = () => {
            if (wsTimeout) clearTimeout(wsTimeout);
            // WebSocket closed - but button state is managed by job tracking
            console.log('WebSocket connection closed');
          };

        } else if (data.image_data_url) {
          // Fallback for old system
          const truncatedPrompt = prompt.substring(0, 900);
          const encodedPrompt = encodeURIComponent(truncatedPrompt).replace(/%20/g, '-');
          const findSimilarBtn = (prompt && prompt.trim() && prompt.trim() !== '.') ?
            `<a href="https://imgsearch.com/search/${encodedPrompt}?ref=freegen" target="_blank" rel="nofollow" class="btn btn-outline-secondary mt-2 me-2 simple-btn">Find Similar</a>` :
            '';
          imageContainer.innerHTML = `
            <img src="${data.image_data_url}" alt="Generated Image" class="img-fluid">
            <br>
            ${findSimilarBtn}
            <a href="${data.image_data_url}" download="generated_image.jpg" class="btn btn-primary mt-2">Download Image</a>
          `;

          // Record completion timing metrics for fallback system
          const totalTime = Date.now() - jobStartTime;
          console.log('Recording completion time (fallback):', totalTime, 'ms');
          fetch(`${STATS_API_URL}/record-completion`, {
            method: 'POST',
            headers: {'Content-Type': 'application/json'},
            body: JSON.stringify({
              job_id: 'fallback_' + Date.now(), // Generate a fallback job ID
              total_time_ms: totalTime,
              timestamp: new Date().toISOString()
            })
          })
          .then(res => res.json())
          .then(result => console.log('Timing recorded (fallback):', result))
          .catch(err => console.error('Failed to record timing metrics (fallback):', err));

          // Mark job as completed and sync button state for non-WebSocket responses
          jobCompleted = true;
          currentJobId = null;
          syncButtonState();
        } else {
          imageContainer.innerHTML = '<div class="text-danger">Failed to generate image.</div>';

          // Mark job as completed and sync button state for non-WebSocket responses
          jobCompleted = true;
          currentJobId = null;
          syncButtonState();
        }
      } catch (err) {
          if (err.message === "Failed to fetch") {
            errorContainer.textContent = 'Too many requests. Please wait a moment and try again.';
          } else {
            errorContainer.textContent = err.message;
          }
          imageContainer.innerHTML = '';

          // Mark job as completed and sync button state on error
          jobCompleted = true;
          currentJobId = null;
          syncButtonState();
        }
    }

    if (generateBtn) {
      generateBtn.addEventListener('click', generateImage);
    }

    if (promptInput) {
      promptInput.addEventListener('keydown', function (event) {
        if (event.key === 'Enter') {
          event.preventDefault();
          if (generateBtn) {
            generateBtn.click();
          }
        }
      });
    }

    function showAdModal() {
      const adModal = new bootstrap.Modal(document.getElementById('adModal'));
      const skipAdBtn = document.getElementById('skipAdBtn');
      const skipTimer = document.getElementById('skipTimer');

      let countdown = 5;
      skipAdBtn.disabled = true;

      // Start countdown
      const countdownInterval = setInterval(() => {
        countdown--;
        skipTimer.textContent = countdown;

        if (countdown <= 0) {
          clearInterval(countdownInterval);
          skipAdBtn.disabled = false;
          skipAdBtn.innerHTML = 'Skip Ad';
        }
      }, 1000);

      // Skip button click handler
      skipAdBtn.addEventListener('click', function() {
        clearInterval(countdownInterval);
        adModal.hide();
        // Reset for next time
        countdown = 5;
        skipTimer.textContent = '5';
        skipAdBtn.innerHTML = 'Skip in <span id="skipTimer">5</span>s';
      });

      adModal.show();
    }

    function incrementGenerationCount() {
      // Only track and show ads if AD_THRESHOLD is greater than 0
      if (AD_THRESHOLD <= 0) {
        return; // Ads disabled
      }

      generationCount++;
      localStorage.setItem('generationCount', generationCount.toString());

      // Show ad after every AD_THRESHOLD generations
      if (generationCount % AD_THRESHOLD === 0) {
        // Delay ad slightly to let user see the generated image first
        setTimeout(() => {
          showAdModal();
        }, 2000);
      }
    }
